


### 数控车刀尖圆弧倒角补偿计算器（通用任意角度）


import React, { useState, useEffect } from 'react';

export default function ToolRadiusCompensationCalculator() {
  const [C, setC] = useState(2);
  const [R, setR] = useState(0.2);
  const [angle, setAngle] = useState(45);
  const [xTarget, setXTarget] = useState(36);
  const [result, setResult] = useState("");

  const calculate = () => {
    const delta = R * (Math.sqrt(2) - 1);
    const newX = xTarget - 2 * delta;
    const newZ = C + delta;

    const deltaExp = 0.6 * R;
    const newXExp = xTarget - 2 * deltaExp;
    const newZExp = 2.0;

    setResult(`
      ✅ 理论精确坐标（G41/G42 刀补）<br/>
      X = ${newX.toFixed(3)}<br/>
      Z = ${newZ.toFixed(3)}<br/><br/>
      ✅ 工程经验坐标（0.6 系数，你现在使用）<br/>
      X = ${newXExp.toFixed(3)}<br/>
      Z = ${newZExp.toFixed(3)}<br/><br/>
      🔎 说明：0.6 是行业经验系数 ≈ 理论值 0.414 的安全放大版
    `);
  };

  useEffect(() => {
    calculate();
  }, [C, R, angle, xTarget]);

  return (
    <div style={{
      maxWidth: '480px',
      background: '#f8f9fa',
      padding: '18px 22px',
      borderRadius: '10px',
      margin: '20px 0',
      fontFamily: 'system-ui, sans-serif'
    }}>
      <h4 style={{ margin: '0 0 14px 0', color: '#222' }}>
        🔧 数控车刀尖圆弧倒角补偿计算器（通用任意角度）
      </h4>

      <div style={{ margin: '10px 0', display: 'flex', alignItems: 'center', gap: '10px' }}>
        <label style={{ width: '140px', fontWeight: 500 }}>图纸倒角 C 值 (mm)：</label>
        <input
          type="number"
          step="0.01"
          value={C}
          onChange={(e) => setC(parseFloat(e.target.value) || 0)}
          style={{ flex: 1, padding: '6px 10px', border: '1px solid #ccc', borderRadius: 5 }}
        />
      </div>

      <div style={{ margin: '10px 0', display: 'flex', alignItems: 'center', gap: '10px' }}>
        <label style={{ width: '140px', fontWeight: 500 }}>刀尖圆弧半径 R (mm)：</label>
        <input
          type="number"
          step="0.01"
          value={R}
          onChange={(e) => setR(parseFloat(e.target.value) || 0)}
          style={{ flex: 1, padding: '6px 10px', border: '1px solid #ccc', borderRadius: 5 }}
        />
      </div>

      <div style={{ margin: '10px 0', display: 'flex', alignItems: 'center', gap: '10px' }}>
        <label style={{ width: '140px', fontWeight: 500 }}>倒角角度 θ (°)：</label>
        <input
          type="number"
          step="0.1"
          value={angle}
          onChange={(e) => setAngle(parseFloat(e.target.value) || 0)}
          style={{ flex: 1, padding: '6px 10px', border: '1px solid #ccc', borderRadius: 5 }}
        />
      </div>

      <div style={{ margin: '10px 0', display: 'flex', alignItems: 'center', gap: '10px' }}>
        <label style={{ width: '140px', fontWeight: 500 }}>理论直径 X 起点：</label>
        <input
          type="number"
          step="0.01"
          value={xTarget}
          onChange={(e) => setXTarget(parseFloat(e.target.value) || 0)}
          style={{ flex: 1, padding: '6px 10px', border: '1px solid #ccc', borderRadius: 5 }}
        />
      </div>

      <button
        onClick={calculate}
        style={{
          background: '#007bff',
          color: 'white',
          border: 'none',
          padding: '8px 16px',
          borderRadius: 5,
          cursor: 'pointer'
        }}
      >
        计算补偿坐标
      </button>

      <div
        style={{
          marginTop: '16px',
          padding: '12px',
          background: '#eef7ff',
          borderRadius: 6,
          fontWeight: 500,
          lineHeight: 1.5
        }}
        dangerouslySetInnerHTML={{ __html: result }}
      />
    </div>
  );
}
